80  	 * Metodo que nos permite comprobar si las cantidades de ingredientes que contiene cada plato,
81  	 * que forma la comanda esta disponible en el alamacén para poder realizar los platos
82  	 * @return boolean
83  	 */
84  	public static boolean consultarDisponibilidadComanda(ArrayList<Ingrediente> ingredientesComanda) throws IOException {
85  		ArrayList<Ingrediente> almacen = new ArrayList<Ingrediente>();
86  
87  		for (Ingrediente i : ingredientesComanda) {
88  			almacen.add(IngredienteDAO.getIngrediente(i.getIdIngrediente(), 1));
89  		}
90  
91  		System.out.println("Los ingredientes requeridos para la comanda son\n");
92  		for (int i = 0; i < ingredientesComanda.size(); i++) {
93  			System.out.println("\t" + ingredientesComanda.get(i).getNombre() + " " +
94  					ingredientesComanda.get(i).getCantidad() + "- " + almacen.get(i).getCantidad());
95  			if (ingredientesComanda.get(i).getCantidad() > almacen.get(i).getCantidad())
96  				return false;
97  		}
98  
99  		return true;
100 	}

181 	/**
182 	 * Metodo que nos inserta todos los ingredientes de todos los platos de una comanda para posteriormente
183 	 * consultar la disponibilidad
184 	 * @param ingredientes
185 	 * @return ArrayList()
186 	 */
187 	public ArrayList<Ingrediente> getIngredientesComanda(ArrayList<Ingrediente> ingredientes) {
188 		ArrayList<Ingrediente> ingredientesComandas = new ArrayList<Ingrediente>();
189 		Collections.sort(ingredientes);
190 
191 		
192 		while(!ingredientes.isEmpty()) {
193 			int rep = Collections.frequency(ingredientes, ingredientes.get(0));
194 			for(int i = 1; i < rep;i++) {
195 				ingredientes.get(0).setCantidad(ingredientes.get(0).getCantidad()+ingredientes.get(1).getCantidad());
196 				ingredientes.remove(1);
197 			}
198 			ingredientesComandas.add(new Ingrediente(ingredientes.get(0).getIdIngrediente(), 
199 					ingredientes.get(0).getNombre(), ingredientes.get(0).getCantidad()));	
200 			ingredientes.remove(0);
201 		}
202 		
203 		return ingredientesComandas;
204 	}
